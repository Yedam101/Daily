```python
import sys


n, s = map(int, input().split())

for i in range(n):
    sib.append(abs(place[i] - s))

def div(n):
    data = []
    for i in range(1, n//2+1):
        if n % i == 0:
            data.append(i)
    data.append(n)
    return list(reversed(data))

l_div = div(min(sib))
for i in l_div:
    check = 0
    for j in sib:
        if j % i != 0:
            check = 1
            break
        else:
            continue
    if check == 0:
        print(i)
        break
```

```python
import sys

n, s = map(int, input().split())
sib_n = list(map(int, sys.stdin.readline().split()))
sib = [abs(i-s) for i in sib_n]

def gcd(a, b):
    while b > 0:
        a, b = b, a % b
    return a

if n == 1:
    print(*sib)
elif n == 2:
    print(gcd(sib[0], sib[1]))
else:
    G = gcd(sib[0], sib[1])
    for i in range(len(sib[2:])):
        G = gcd(G, sib[i])
    print(G)
```

```python
# 히히 성공

import sys
def era(n):
    sieve = [False, False] + [True]*(n-1)
    for i in range(2, n+1):
        if sieve[i] == True:
            for j in range(i*2, n+1, i):
                sieve[j] = False
    return sieve

t = int(sys.stdin.readline())
for i in range(t):
    count = 0
    n = int(sys.stdin.readline())
    a = 0
    b = n
    sieve = era(n)
    for i in range(len(sieve)):
        if sieve[a] and sieve[b]:
            count += 1
        a += 1
        b -= 1
        if a > b:
            break
    print(count)   

```

```python
# 훨씬 효율적인 남의 답

def get_primary_list(n):
    array = [1 for _ in range(n+1)]
    for i in range(2, int(n ** 0.5) + 1):
        if array[i]:
            j = 2
        while i * j <= n:
            array[i * j] = 0
            j += 1
    return array

T = int(input())
nums = [int(input()) for _ in range(T)]
max_num = max(nums)
primarys = get_primary_list(max_num)

for num in nums:
    result = 0
    for i in range(2, num // 2 + 1):
        if primarys[i] and primarys[num - i]:
            result += 1

    print(result) 

```

## 에라토스테네스의 체 효율적인 소스코드

```python
def era(n):
    sieve = [False, False] + [True] * (n-1)
    for i in range(2, int(n ** 0.5) + 1):
        if sieve[i] == True:         
            for j in range(2*i, n+1, i):
                sieve[j] = False
    return [i for i in range(2, n+1) if sieve[i] == True]

```
