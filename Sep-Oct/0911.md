```python
# 시간초과

N = int(input())
n = list(map(int, input().split()))
answer = []
stack = []
max_m = -1

for i in range(len(n)):
    stack.append(n[i])
    if i == 0 or n[i] > stack[i-1]:
        check = 0
        point = i + 1
        while point < len(n):
            if n[point] > stack[i]:
                answer.append(n[point])
                max_m = n[point]
                check = 1
                break
            else:
                point += 1
        if check == 0:
            answer.append(-1)
    elif i == len(n) - 1:
        answer.append(-1)
    elif n[i] <= stack[-1]:
        answer.append(max_m)

        
print(*answer, sep=' ')

```

```python
for i in range(len(n)):
    stack = n[i:]
    m = stack.pop(0)
    if not stack or m >= max(stack):
        answer.append(-1)
    elif m < stack[0]:
        answer.append(stack[0])
    else:
        answer.append(max(stack))
```

## 최대공약수와 최소공배수
```python
# GCD, LCM 구하는 법 암기

def gcd(a, b):
    while b > 0:
        a, b =  b, a%b
    return a

def lcm(a, b):
    return a*b / gcd(a, b)

a, b = map(int, input().split())
print(gcd(a, b))
print(int(lcm(a, b)))    

```

```python
# 소수 판별법 외우기

n = int(input())
N = list(map(int, input().split()))
count = 0

def prime(x):
    if x == 1:
        return False
    for i in range(2, int(x**(1/2) + 1)):
        if x % i == 0:
            return False 
    return True 

for i in N:
    if prime(i):
        count += 1
        
print(count)

```

```python
# 에라토스테네스의 체 암기

def era(a, b):
    seive = [False, False] + [True]*(b-1)
    
    for i in range(2, b+1):
        if seive[i] == True:
            for j in range(i*2, b+1, i):
                seive[j] = False
    return [i for i in range(a, b+1) if seive[i] == True]

n, m = map(int, input().split())

for i in era(n, m):
    print(i)
```

## 골드바흐의 추측
```python
# 시간초과

import sys

def era(n):
    sieve = [True] * n
    m = int(n ** 0.5)
    for i in range(2, m + 1):
        if sieve[i] == True: 
            for j in range(i*2, n, i):
                sieve[j] = False
    return [i for i in range(2, n) if sieve[i] == True]



while True:
    n = int(sys.stdin.readline())
    if n != 0:
        m = era(n)
        left = 0
        right = len(m)- 1

        while left < right:
            if m[left] + m[right] == n:
                print(f'{n} = {m[left]} + {m[right]}')
                break
            elif m[left] + m[right] < n:
                left += 1
            else:
                right -= 1
    else:
        break
  

```

```python
    if array[i]:

            array[k] = False

while True:

    if n == 0: break
```

## GCD 합
```python
# itertools의 combinations를 이용해 조합을 구한 후 각각의 GCD를 함수를 통해 구함.

from itertools import combinations
import sys

n = int(input())

def gcd(a, b):
    while b > 0:
        a, b = b, a%b
    return a

for i in range(n):
    m = list(map(int, sys.stdin.readline().split()))[1:]
    combi = list(combinations(m, 2))
    count = 0

    for i in combi:
        count += gcd(i[0], i[1])
    print(count)

```
